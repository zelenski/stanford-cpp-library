26 /*
27  * Statement: foreach
28  * Usage: foreach (type var in collection) { ... }
29  * -----------------------------------------------
30  * The <code>foreach</code> statement steps through the elements in
31  * a collection. It works correctly with the collection classes in
32  * both the Standard Template Library and the Stanford C++ libraries,
33  * but can also be used with C++ strings and statically initialized
34  * arrays.
35  *
36  * <p>The following code, for example, prints every element in the
37  * string vector <code>lines</code>:
38  *
39  *<pre>
40  * foreach (string str in lines) {
41  * cout &lt;&lt; str &lt;&lt; endl;
42  * }
43  *</pre>
44  *
45  * Similarly, the following function calculates the sum of the character
46  * codes in a string:
47  *
48  *<pre>
49  * int sumCharacterCodes(string str) {
50  * int sum = 0;
51  * foreach (char ch in str) sum += ch;
52  * return sum;
53  * }
54  *</pre>
55  *
56  * As a simplification when iterating over maps, the <code>foreach</code>
57  * macro iterates through the keys rather than the key/value pairs.
58  */

138 template <typename DowncastType, typename ValueType>
139 ValueType HookImpl(State& fe) {
140  DowncastType *ip = (DowncastType *) fe.itr;
141  if (ip->iter == ip->end) {
142  fe.state = 2;
143  return ValueType();
144  }
145  fe.state = 1;
146  ValueType vp = *ip->iter; /* Subtle implementation note: */
147  ++ip->iter; /* Using *ip->iter++ here would */
148  return vp; /* require copying the iterator. */
149 }

174 template <typename DowncastType, typename ValueType>
175 ValueType MapHookImpl(State & fe) {
176  DowncastType *ip = (DowncastType *) fe.itr;
177  if (ip->iter == ip->end) {
178  fe.state = 2;
179  return ValueType();
180  }
181  fe.state = 1;
182  ValueType key = ip->iter->first;
183  ++ip->iter;
184  return key;
185 }
