2  * File: sparsegrid.h
3  * ------------------
4  * This file exports the <code>SparseGrid</code> class, which offers a
5  * convenient abstraction for representing a two-dimensional array.
6  * Unlike Grid, SparseGrid is optimized for grids in which most of the cells
7  * are empty. It uses far less memory in such cases than a Grid does.
8  * If the grid is expected to be mostly full of meaningful data,
9  * Grid is recommended for use over SparseGrid.
10  *
11  * @author Marty Stepp
12  * @version 2018/03/12
13  * - added overloads that accept GridLocation: get, inBounds, isSet, locations,
14  * set, unset, operator []
15  * @version 2018/03/10
16  * - added methods front, back, clear
17  * @version 2017/11/14
18  * - added iterator version checking support
19  * @version 2016/10/22
20  * - bug fix for size method
21  * @version 2016/09/24
22  * - refactored to use collections.h utility functions
23  * - added size() method
24  * @version 2016/08/10
25  * - added constructor support for std initializer_list usage, such as
26  * {{1, 2, 3}, {4, 5, 6}}
27  * @version 2016/08/04
28  * - fixed operator >> to not throw errors
29  * @version 2015/07/05
30  * - using global hashing functions rather than global variables

440  * Deep copying support
441  * --------------------
442  * This copy constructor and operator= are defined to make a
443  * deep copy, making it possible to pass/return grids by value
444  * and assign from one grid to another. The entire contents of
445  * the grid, including all elements, are copied. Each grid
446  * element is copied from the original grid to the copy using
447  * assignment (operator=). Making copies is generally avoided

453  elements = grid.elements;
454  nRows = grid.nRows;

636 template <typename ValueType>
637 SparseGrid<ValueType>::SparseGrid(std::initializer_list<std::initializer_list<ValueType> > list)
638  : nRows(0),
639  nCols(0) {
640  // create the grid at the proper size
641  nRows = list.size();
642  if (list.begin() != list.end()) {
643  nCols = list.begin()->size();
644  }
645  resize(nRows, nCols);
646 
647  // copy the data from the initializer list into the Grid
648  auto rowItr = list.begin();
649  for (int row = 0; row < nRows; row++) {
650  if ((int) rowItr->size() != nCols) {
651  error("SparseGrid::constructor: initializer list is not rectangular (must have same # cols in each row)");
652  }
653  auto colItr = rowItr->begin();
654  for (int col = 0; col < nCols; col++) {
655  set(row, col, *colItr);
656  colItr++;
657  }
658  rowItr++;
659  }
660 }

683 template <typename ValueType>
684 bool SparseGrid<ValueType>::equals(const SparseGrid<ValueType>& grid2) const {
685  // optimization: if literally same grid, stop
686  if (this == &grid2) {
687  return true;
688  }
689  if (nRows != grid2.nRows || nCols != grid2.nCols) {
690  return false;
691  }
692  for (int row = 0; row < nRows; row++) {
693  for (int col = 0; col < nCols; col++) {
694  if (isSet(row, col)) {
695  // I have data there; he must, too, and it must be the same data
696  if (!grid2.isSet(row, col) || get(row, col) != grid2.get(row, col)) {
697  return false;
698  }
699  } else {
700  // I don't have data there; he must also not have it there
701  if (grid2.isSet(row, col)) {
702  return false;
703  }
704  }
705  }
706  }
707  return true;
708 }

859  }
860  } else {

972 template <typename ValueType>
973 int SparseGrid<ValueType>::gridCompare(const SparseGrid& grid2) const {
974  int h1 = height();
975  int w1 = width();
976  int h2 = grid2.height();
977  int w2 = grid2.width();
978  int rows = h1 > h2 ? h1 : h2;
979  int cols = w1 > w2 ? w1 : w2;
980  for (int r = 0; r < rows; r++) {
981  for (int c = 0; c < cols; c++) {
982  if (r >= h1) {
983  return -1;
984  } else if (r >= h2) {
985  return 1;
986  }
987 
988  if (c >= w1) {
989  return -1;
990  } else if (c >= w2) {
991  return 1;
992  }
993 
994  if (!isSet(r, c) && grid2.isSet(r, c)) {
995  return -1;
996  } else if (isSet(r, c) && !grid2.isSet(r, c)) {
997  return 1;
998  }
999 
1000  if (get(r, c) < grid2.get(r, c)) {
1001  return -1;
1002  } else if (grid2.get(r, c) < get(r, c)) {
1003  return 1;
1004  }
1005  }
1006  }
1007  return 0;
1008 }
