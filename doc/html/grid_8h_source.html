59 /*
60  * Class: Grid<ValueType>
61  * ----------------------
62  * This class stores an indexed, two-dimensional array. The following code,
63  * for example, creates an identity matrix of size <code>n</code>, in which
64  * the elements are 1.0 along the main diagonal and 0.0 everywhere else:
65  *
66  *<pre>
67  * Grid&lt;double&gt; createIdentityMatrix(int n) {
68  * Grid&lt;double&gt; matrix(n, n);
69  * for (int i = 0; i &lt; n; i++) {
70  * matrix[i][i] = 1.0;
71  * }
72  * return matrix;
73  * }
74  *</pre>
75  */

437  /*
438  * Deep copying support
439  * --------------------
440  * This copy constructor and operator= are defined to make a
441  * deep copy, making it possible to pass/return grids by value
442  * and assign from one grid to another. The entire contents of
443  * the grid, including all elements, are copied. Each grid
444  * element is copied from the original grid to the copy using
445  * assignment (operator=). Making copies is generally avoided
446  * because of the expense and thus, grids are typically passed
447  * by reference, however, when a copy is needed, these operations
448  * are supported.
449  */
450  void deepCopy(const Grid& grid) {
451  int n = grid.nRows * grid.nCols;
452  elements = new ValueType[n];
453  for (int i = 0; i < n; i++) {
454  elements[i] = grid.elements[i];
455  }
456  nRows = grid.nRows;
457  nCols = grid.nCols;
458  m_version++;
459  }
460

462  Grid& operator =(const Grid& src) {
463  if (this != &src) {
464  delete[] elements;
465  deepCopy(src);
466  }
467  return *this;
468  }

641 template <typename ValueType>
642 Grid<ValueType>::Grid(std::initializer_list<std::initializer_list<ValueType> > list)
643  : elements(nullptr),
644  nRows(0),
645  nCols(0) {
646  // create the grid at the proper size
647  nRows = list.size();
648  if (list.begin() != list.end()) {
649  nCols = list.begin()->size();
650  }
651  resize(nRows, nCols);
652 
653  // copy the data from the initializer list into the Grid
654  auto rowItr = list.begin();
655  for (int row = 0; row < nRows; row++) {
656  if ((int) rowItr->size() != nCols) {
657  error("Grid::constructor: initializer list is not rectangular (must have same # cols in each row)");
658  }
659  auto colItr = rowItr->begin();
660  for (int col = 0; col < nCols; col++) {
661  set(row, col, *colItr);
662  colItr++;
663  }
664  rowItr++;
665  }
666 }

694 template <typename ValueType>
695 bool Grid<ValueType>::equals(const Grid<ValueType>& grid2) const {
696  // optimization: if literally same grid, stop
697  if (this == &grid2) {
698  return true;
699  }
700 
701  if (nRows != grid2.nRows || nCols != grid2.nCols) {
702  return false;
703  }
704  for (int row = 0; row < nRows; row++) {
705  for (int col = 0; col < nCols; col++) {
706  if (get(row, col) != grid2.get(row, col)) {
707  return false;
708  }
709  }
710  }
711  return true;
712 }

844 template <typename ValueType>
845 void Grid<ValueType>::resize(int numRows, int numCols, bool retain) {
846  if (numRows < 0 || numCols < 0) {
847  std::ostringstream out;
848  out << "Grid::resize: Attempt to resize grid to invalid size ("
849  << numRows << ", " << numCols << ")";
850  error(out.str());
851  }
852 
853  // optimization: don't do the resize if we are already that size
854  if (numRows == this->nRows && numCols == this->nCols && retain) {
855  return;
856  }
857 
858  // save backup of old array/size
859  ValueType* oldElements = this->elements;
860  int oldnRows = this->nRows;
861  int oldnCols = this->nCols;
862 
863  // create new empty array and set new size
864  this->nRows = numRows;
865  this->nCols = numCols;
866  this->elements = new ValueType[numRows * numCols];
867 
868  // initialize to empty/default state
869  ValueType value = ValueType();
870  for (int i = 0; i < numRows * numCols; i++) {
871  this->elements[i] = value;
872  }
873 
874  // possibly retain old contents
875  if (retain) {
876  int minRows = oldnRows < numRows ? oldnRows : numRows;
877  int minCols = oldnCols < numCols ? oldnCols : numCols;
878  for (int row = 0; row < minRows; row++) {
879  for (int col = 0; col < minCols; col++) {
880  this->elements[(row * numCols) + col] = oldElements[(row * oldnCols) + col];
881  }
882  }
883  }
884 
885  // free old array memory
886  if (oldElements) {
887  delete[] oldElements;
888  }
889  m_version++;
890 }
