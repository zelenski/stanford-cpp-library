4  * This file patches the implementation of the random number library
5  * to avoid some serious bugs in standard implementations of rand,
6  * particularly on Mac OS X. It also includes a hack to set the
7  * seed from the RANDOM_SEED environment variable, which makes it

11 /*
12  * Implementation notes: rand, srand
13  * ---------------------------------
14  * To ensure that this package works the same way on all platforms,
15  * this file completely reimplements the rand and srand methods. The
16  * algorithm is a conventional linear congruential generator with the
17  * parameters used in GNU's gclib. RAND_MAX for this implementation
18  * is 2147483647 (2^31 - 1).
19  */
20 
21 #define MULTIPLIER 1103515245
22 #define OFFSET 12345
23 
24 static int _seed = 1;
25 
26 #undef rand
27 #define rand() ((_seed = MULTIPLIER * _seed + OFFSET) & 0x7FFFFFFF)
28 
29 #undef srand
30 #define srand(seed) (_seed = int(seed), _seed = (_seed <= 0) ? 1 : _seed)
31 
32 #undef RAND_MAX
33 #define RAND_MAX 2147483647

35 /*
36  * Implementation notes: Windows patch
37  * -----------------------------------
38  * On some versions of Windows, the time function is too coarse to use

41  */

43 #if defined (_MSC_VER) && (_MSC_VER >= 1200)
