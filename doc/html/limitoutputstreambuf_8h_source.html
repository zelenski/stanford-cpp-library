22 /*
23  * An output stream buffer that counts the number of characters written to it.
24  * Used to limit cout console output by autograders for naughty students who
25  * print too much output.
26  *
27  * inspired by: http://gabisoft.free.fr/articles/fltrsbf1.html
28  */
29 class LimitOutputStreambuf : public std::streambuf {
30 public:
31  LimitOutputStreambuf(std::streambuf* source, int max) : m_source(source), m_count(0), m_max(max) {
32  // empty
33  }
34 
35  virtual ~LimitOutputStreambuf() {
36  sync();
37  }

39  /*
40  * This is the crucial function; called to write a character to the
41  * underlying stream buffer (cout). We count them so we can throw
42  * an error if too many are printed.
43  */
44  virtual int overflow(int ch) {
45  if (m_count >= 0) {
46  m_count++;
47  if (m_count > m_max) {
48  m_count = -1; // disable checking on further calls so I can print again
49  std::ostringstream os;
50  os << std::endl;
51  os << "*** ERROR: Excessive output produced! (over " << m_max << " chars)" << std::endl;
52  os << "*** Halting program." << std::endl;
53  std::string str = os.str();
54  m_source->sputn(str.c_str(), (int) str.length());
55  throw std::exception();
56  }
57  }
58 
59  return m_source->sputc(ch);
60  }
61 
62  virtual int underflow() {
63  return EOF;
64  }
65 
66  virtual int sync() {
67  return m_source->pubsync();
68  }
69 
70  virtual std::streambuf* setbuf(char* p, std::streamsize len) {
71  return m_source->pubsetbuf(p, len);
72  }
73 
74 private:
75  std::streambuf* m_source;
