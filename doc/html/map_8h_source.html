447  * Implementation notes:
448  * ---------------------
449  * The map class is represented using a binary search tree. The
450  * specific implementation used here is the classic AVL algorithm
451  * developed by Georgii Adel'son-Vel'skii and Evgenii Landis in 1962.

458  static const int BST_RIGHT_HEAVY = +1;

460  /* Type definition for nodes in the binary search tree */
461  struct BSTNode {
462  KeyType key; /* The key stored in this node */
463  ValueType value; /* The corresponding value */
464  BSTNode* left; /* Subtree containing all smaller keys */
465  BSTNode* right; /* Subtree containing all larger keys */
466  int bf; /* AVL balance factor */
467  };

469  /*
470  * Implementation notes: Comparator
471  * --------------------------------
472  * The Comparator class encapsulates a functor that compares two values
473  * of KeyType. In contrast to the classes in the STL, all of which embed
474  * the comparator in the type, the Map class and its derivatives pass an
475  * optional Comparator value. While this strategy results in a more
476  * complex implementation, it has the advantage of allowing maps and sets
477  * to carry their own comparators without forcing the client to include
478  * the comparator in the template declaration. This simplification is
479  * particularly important for the Graph class.
480  *
481  * The allocation is required in the TemplateComparator class because
482  * the type std::binary_function has subclasses but does not define a
483  * virtual destructor.
484  */
485  class Comparator {
486  public:
487  virtual ~Comparator() { /* empty */ }
488  virtual bool lessThan(const KeyType& k1, const KeyType& k2) = 0;
489  virtual Comparator* clone() = 0;
490  };

577  /*
578  * Implementation notes: findNode(t, key)
579  * --------------------------------------
580  * Searches the tree rooted at t to find the specified key, searching
581  * in the left or right subtree, as approriate. If a matching node
582  * is found, findNode returns a pointer to the value cell in that node.
583  * If no matching node exists in the tree, findNode returns nullptr.
584  */
585  ValueType* findNode(BSTNode* t, const KeyType& key) const {
586  if (!t) {
587  return nullptr;
588  }
589  int sign = compareKeys(key, t->key);
590  if (sign == 0) {
591  return &t->value;
592  }
593  if (sign < 0) {
594  return findNode(t->left, key);
595  } else {
596  return findNode(t->right, key);
597  }
598  }
599 
600  /*
601  * Implementation notes: addNode(t, key, heightFlag)
602  * -------------------------------------------------
603  * Searches the tree rooted at t to find the specified key, searching
604  * in the left or right subtree, as approriate. If a matching node
605  * is found, addNode returns a pointer to the value cell in that node,
606  * just like findNode. If no matching node exists in the tree, addNode
607  * creates a new node with a default value. The heightFlag reference
608  * parameter returns a bool indicating whether the height of the tree
609  * was changed by this operation.
610  */
611  ValueType* addNode(BSTNode*& t, const KeyType& key, bool& heightFlag) {
612  heightFlag = false;
613  if (!t) {
614  t = new BSTNode();
615  t->key = key;
616  t->value = ValueType();
617  t->bf = BST_IN_BALANCE;
618  t->left = t->right = nullptr;
619  heightFlag = true;
620  nodeCount++;
621  return &t->value;
622  }
623 
624  int sign = compareKeys(key, t->key);
625  if (sign == 0) {
626  return &t->value;
627  }
628  ValueType* vp = nullptr;
629  int bfDelta = BST_IN_BALANCE;
630  if (sign < 0) {
631  vp = addNode(t->left, key, heightFlag);
632  if (heightFlag) {
633  bfDelta = BST_LEFT_HEAVY;
634  }
635  } else {
636  vp = addNode(t->right, key, heightFlag);
637  if (heightFlag) {
638  bfDelta = BST_RIGHT_HEAVY;
639  }
640  }
641  updateBF(t, bfDelta);
642  heightFlag = (bfDelta != 0 && t->bf != BST_IN_BALANCE);
643  return vp;
644  }

646  /*
647  * Implementation notes: removeNode(t, key)
648  * ----------------------------------------
649  * Removes the node containing the specified key from the tree rooted
650  * at t. The return value is true if the height of this subtree
651  * changes. The removeTargetNode method does the actual deletion.
652  */
653  bool removeNode(BSTNode*& t, const KeyType& key) {
654  if (!t) {
655  return false;
656  }
657  int sign = compareKeys(key, t->key);
658  if (sign == 0) {
659  return removeTargetNode(t);
660  }
661  int bfDelta = BST_IN_BALANCE;
662  if (sign < 0) {
663  if (removeNode(t->left, key)) {
664  bfDelta = BST_RIGHT_HEAVY;
665  }
666  } else {
667  if (removeNode(t->right, key)) {
668  bfDelta = BST_LEFT_HEAVY;
669  }
670  }
671  updateBF(t, bfDelta);
672  return bfDelta != 0 && t->bf == BST_IN_BALANCE;
673  }

675  /*
676  * Implementation notes: removeTargetNode(t)
677  * -----------------------------------------
678  * Removes the node which is passed by reference as t. The easy case
679  * occurs when either (or both) of the children is null; all you need
680  * to do is replace the node with its non-null child, if any. If both
681  * children are non-null, this code finds the rightmost descendent of
682  * the left child; this node may not be a leaf, but will have no right
683  * child. Its left child replaces it in the tree, after which the
684  * replacement data is moved to the position occupied by the target node.
685  */
686  bool removeTargetNode(BSTNode*& t) {
687  BSTNode* toDelete = t;
688  if (!t->left) {
689  t = t->right;
690  delete toDelete;
691  nodeCount--;
692  return true;
693  } else if (!t->right) {
694  t = t->left;
695  delete toDelete;
696  nodeCount--;
697  return true;
698  } else {
699  BSTNode* successor = t->left;
700  while (successor->right) {
701  successor = successor->right;
702  }
703  t->key = successor->key;
704  t->value = successor->value;
705  if (removeNode(t->left, successor->key)) {
706  updateBF(t, BST_RIGHT_HEAVY);
707  return (t->bf == BST_IN_BALANCE);
708  }
709  return false;
710  }
711  }
712 
713  /*
714  * Implementation notes: updateBF(t, bfDelta)
715  * ------------------------------------------
716  * Updates the balance factor in the node and rebalances the tree
717  * if necessary.
718  */
719  void updateBF(BSTNode*& t, int bfDelta) {
720  t->bf += bfDelta;
721  if (t->bf < BST_LEFT_HEAVY) {
722  fixLeftImbalance(t);
723  } else if (t->bf > BST_RIGHT_HEAVY) {
724  fixRightImbalance(t);
725  }
726  }
727 
728  /*
729  * Implementation notes: fixLeftImbalance(t)
730  * -----------------------------------------
731  * This function is called when a node has been found that is out
732  * of balance with the longer subtree on the left. Depending on
733  * the balance factor of the left child, the code performs a
734  * single or double rotation.
735  */
736  void fixLeftImbalance(BSTNode*& t) {
737  BSTNode *child = t->left;
738  if (child->bf == BST_RIGHT_HEAVY) {
739  int oldBF = child->right->bf;
740  rotateLeft(t->left);
741  rotateRight(t);
742  t->bf = BST_IN_BALANCE;
743  switch (oldBF) {
744  case BST_LEFT_HEAVY:
745  t->left->bf = BST_IN_BALANCE;
746  t->right->bf = BST_RIGHT_HEAVY;
747  break;
748  case BST_IN_BALANCE:
749  t->left->bf = t->right->bf = BST_IN_BALANCE;
750  break;
751  case BST_RIGHT_HEAVY:
752  t->left->bf = BST_LEFT_HEAVY;
753  t->right->bf = BST_IN_BALANCE;
754  break;
755  }
756  } else if (child->bf == BST_IN_BALANCE) {
757  rotateRight(t);
758  t->bf = BST_RIGHT_HEAVY;
759  t->right->bf = BST_LEFT_HEAVY;
760  } else {
761  rotateRight(t);
762  t->right->bf = t->bf = BST_IN_BALANCE;
763  }
764  }
765 
766  /*
767  * Implementation notes: rotateLeft(t)
768  * -----------------------------------
769  * This function performs a single left rotation of the tree
770  * that is passed by reference. The balance factors
771  * are unchanged by this function and must be corrected at a
772  * higher level of the algorithm.
773  */
774  void rotateLeft(BSTNode*& t) {
775  BSTNode* child = t->right;
776  t->right = child->left;
777  child->left = t;
778  t = child;
779  }
780 
781  /*
782  * Implementation notes: fixRightImbalance(t)
783  * ------------------------------------------
784  * This function is called when a node has been found that
785  * is out of balance with the longer subtree on the right.
786  * Depending on the balance factor of the right child, the
787  * code performs a single or double rotation.
788  */
789  void fixRightImbalance(BSTNode*& t) {
790  BSTNode* child = t->right;
791  if (child->bf == BST_LEFT_HEAVY) {
792  int oldBF = child->left->bf;
793  rotateRight(t->right);
794  rotateLeft(t);
795  t->bf = BST_IN_BALANCE;
796  switch (oldBF) {
797  case BST_LEFT_HEAVY:
798  t->left->bf = BST_IN_BALANCE;
799  t->right->bf = BST_RIGHT_HEAVY;
800  break;
801  case BST_IN_BALANCE:
802  t->left->bf = t->right->bf = BST_IN_BALANCE;
803  break;
804  case BST_RIGHT_HEAVY:
805  t->left->bf = BST_LEFT_HEAVY;
806  t->right->bf = BST_IN_BALANCE;
807  break;
808  }
809  } else if (child->bf == BST_IN_BALANCE) {
810  rotateLeft(t);
811  t->bf = BST_LEFT_HEAVY;
812  t->left->bf = BST_RIGHT_HEAVY;
813  } else {
814  rotateLeft(t);
815  t->left->bf = t->bf = BST_IN_BALANCE;
816  }
817  }
818 
819  /*
820  * Implementation notes: rotateRight(t)
821  * ------------------------------------
822  * This function performs a single right rotation of the tree
823  * that is passed by reference. The balance factors
824  * are unchanged by this function and must be corrected at a
825  * higher level of the algorithm.
826  */
827  void rotateRight(BSTNode*& t) {
828  BSTNode* child = t->left;
829  t->left = child->right;
830  child->right = t;
831  t = child;
832  }

965  private:

976  void findLeftmostChild() {
977  BSTNode *np = stack.peek().np;
978  if (!np) {
979  return;
980  }
981  while (np->left) {
982  NodeMarker marker = { np->left, false };
983  stack.push(marker);
984  np = np->left;
985  }

1023  stack.pop();
