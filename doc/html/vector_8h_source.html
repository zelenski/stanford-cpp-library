847 void Vector<ValueType>::ensureCapacity(int cap) {
848  if (cap >= 1 && capacity < cap) {
849  capacity = std::max(cap, capacity * 2);
850  ValueType* array = new ValueType[capacity];
851  if (elements) {
852  for (int i = 0; i < count; i++) {
853  array[i] = elements[i];
854  }
855  delete[] elements;
856  }
857  elements = array;
858  }

919  * Implementation notes: insert, remove, add
920  * -----------------------------------------
921  * These methods must shift the existing elements in the array to

923  * deleted one.

925 template <typename ValueType>
926 void Vector<ValueType>::insert(int index, const ValueType& value) {
927  checkIndex(index, 0, count, "insert");
928  if (count == capacity) {
929  expandCapacity();
930  }
931  for (int i = count; i > index; i--) {
932  elements[i] = elements[i - 1];
933  }
934  elements[index] = value;
935  count++;
936  m_version++;
937 }
