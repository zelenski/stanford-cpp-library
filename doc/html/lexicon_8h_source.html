469  struct TrieNode {
470  public:
471  TrieNode() {
472  m_isWord = false;
473  for (int i = 0; i < 26; i++) {
474  m_children[i] = nullptr;
475  }
476  }
477 
478  // pre: letter is between 'a' and 'z' in lowercase
479  inline TrieNode*& child(char letter) {
480  return m_children[letter - 'a'];
481  }
482 
483  inline int childCount() const {
484  int count = 0;
485  for (int i = 0; i < 26; i++) {
486  if (m_children[i] != nullptr) {
487  count++;
488  }
489  }
490  return count;
491  }
492 
493  inline bool isLeaf() const {
494  return childCount() == 0;
495  }
