37 /*
38  * Class: Lexicon
39  * --------------
40  * This class is used to represent a <b><i>lexicon,</i></b> or word list.
41  * The main difference between a lexicon and a dictionary is that
42  * a lexicon does not provide any mechanism for storing definitions;
43  * the lexicon contains only words, with no associated information.
44  * It is therefore similar to a set of strings, but with a more
45  * space-efficient internal representation. The <code>Lexicon</code>
46  * class supports efficient lookup operations for words and prefixes.
47  *
48  * <p>As an example of the use of the <code>Lexicon</code> class, the
49  * following program lists all the two-letter words in the lexicon
50  * stored in <code>EnglishWords.dat</code>:
51  *
52  *<pre>
53  * int main() {
54  * Lexicon english("EnglishWords.dat");
55  * for (string word : english) {
56  * if (word.length() == 2) {
57  * cout << word << endl;
58  * }
59  * }
60  * return 0;
61  * }
62  *</pre>
63  */
64

493  struct TrieNode {
494  public:
495  TrieNode() {
496  m_isWord = false;
497  for (int i = 0; i < 26; i++) {
498  m_children[i] = nullptr;
499  }
500  }
501 
502  // pre: letter is between 'a' and 'z' in lowercase
503  inline TrieNode*& child(char letter) {
504  return m_children[letter - 'a'];
505  }
506 
507  inline int childCount() const {
508  int count = 0;
509  for (int i = 0; i < 26; i++) {
510  if (m_children[i] != nullptr) {
511  count++;
512  }
513  }
514  return count;
515  }
516 
517  inline bool isLeaf() const {
518  return childCount() == 0;
519  }
520 
521  inline bool isWord() const {
522  return m_isWord;
523  }
524 
525  inline void setWord(bool value) {
526  m_isWord = value;
527  }
