1209  template <typename C, typename T>
1210  basic_pstreambuf<C,T>*
1211  basic_pstreambuf<C,T>::open( const string& file,
1212  const argv_type& argv,
1213  pmode mode )
1214  {
1215  basic_pstreambuf<C,T>* ret = nullptr;
1216 
1217  if (!is_open())
1218  {
1219  // constants for read/write ends of pipe
1220  enum { RD, WR };
1221 
1222  // open another pipe and set close-on-exec
1223  fd_type ck_exec[] = { -1, -1 };
1224  if (-1 == ::pipe(ck_exec)
1225  || -1 == ::fcntl(ck_exec[RD], F_SETFD, FD_CLOEXEC)
1226  || -1 == ::fcntl(ck_exec[WR], F_SETFD, FD_CLOEXEC))
1227  {
1228  error_ = errno;
1229  close_fd_array(ck_exec);
1230  }
1231  else
1232  {
1233  switch(fork(mode))
1234  {
1235  case 0 :
1236  // this is the new process, exec command
1237  {
1238  char** arg_v = new char*[argv.size()+1];
1239  for (std::size_t i = 0; i < argv.size(); ++i)
1240  {
1241  const string& src = argv[i];
1242  char*& dest = arg_v[i];
1243  dest = new char[src.size()+1];
1244  dest[ src.copy(dest, src.size()) ] = '\0';
1245  }
1246  arg_v[argv.size()] = nullptr;
1247 
1248  ::execvp(file.c_str(), arg_v);
1249 
1250  // can only reach this point if exec() failed
1251 
1252  // parent can get error code from ck_exec pipe
1253  error_ = errno;
1254 
1255  while (::write(ck_exec[WR], &error_, sizeof(error_)) == -1
1256  && errno == EINTR)
1257  { }
1258 
1259  ::close(ck_exec[WR]);
1260  ::close(ck_exec[RD]);
1261 
1262  ::_exit(error_);
1263  // using std::exit() would make static dtors run twice
1264  }
1265 
1266  case -1 :
1267  // couldn't fork, error already handled in pstreambuf::fork()
1268  close_fd_array(ck_exec);
1269  break;
1270 
1271  default :
1272  // this is the parent process
1273 
1274  // check child called exec() successfully
1275  ::close(ck_exec[WR]);
1276  switch (::read(ck_exec[RD], &error_, sizeof(error_)))
1277  {
1278  case 0:
1279  // activate buffers
1280  create_buffers(mode);
1281  ret = this;
1282  break;
1283  case -1:
1284  error_ = errno;
1285  break;
1286  default:
1287  // error_ contains error code from child
1288  // call wait() to clean up and set ppid_ to 0
1289  this->wait();
1290  break;
1291  }
1292  ::close(ck_exec[RD]);
1293  }
1294  }
1295  }
1296  return ret;
1297  }

1322  // process' stdin, stdout and stderr
1323  // (stored in a single array so close_fd_array() can close all at once)
1324  fd_type fd[] = { -1, -1, -1, -1, -1, -1 };

1326  fd_type* const pout = fd+2;
1327  fd_type* const perr = fd+4;
1328 
1329  // constants for read/write ends of pipe
1330  enum { RD, WR };
1331 
1332  // N.B.
1333  // For the pstreambuf pin is an output stream and
1334  // pout and perr are input streams.
1335 
1336  if (!error_ && mode&pstdin && ::pipe(pin))
1337  error_ = errno;
1338 
1339  if (!error_ && mode&pstdout && ::pipe(pout))
1340  error_ = errno;
1341 
1342  if (!error_ && mode&pstderr && ::pipe(perr))
1343  error_ = errno;
1344 
1345  if (!error_)
1346  {
1347  pid = ::fork();
1348  switch (pid)
1349  {
1350  case 0 :
1351  {
1352  // this is the new process
1353 
1354  // for each open pipe close one end and redirect the
1355  // respective standard stream to the other end
1356 
1357  if (*pin >= 0)
1358  {
1359  ::close(pin[WR]);
1360  ::dup2(pin[RD], STDIN_FILENO);
1361  ::close(pin[RD]);
1362  }
1363  if (*pout >= 0)
1364  {
1365  ::close(pout[RD]);
1366  ::dup2(pout[WR], STDOUT_FILENO);
1367  ::close(pout[WR]);
1368  }
1369  if (*perr >= 0)

1372  ::dup2(perr[WR], STDERR_FILENO);
1373  ::close(perr[WR]);
1374  }
1375 
1376 #ifdef _POSIX_JOB_CONTROL
1377  if (mode&newpg)
1378  ::setpgid(0, 0); // Change to a new process group
1379 #endif
1380 
1381  break;
1382  }
1383  case -1 :
1384  {
1385  // couldn't fork for some reason
1386  error_ = errno;
1387  // close any open pipes
1388  close_fd_array(fd);
1389  break;
1390  }
1391  default :
1392  {
1393  // this is the parent process, store process' pid
1394  ppid_ = pid;
1395

1401  }
1402  if (*pout >= 0)
1403  {

1406  }

1410  ::close(perr[WR]);
1411  }
1412  }
1413  }
1414  }

1421  }
