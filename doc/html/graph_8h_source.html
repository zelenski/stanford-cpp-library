769  class GraphComparator {
770  public:
771  bool operator ()(NodeType* n1, NodeType* n2) {
772  return compare(n1, n2) < 0;
773  }

808  static int compare(NodeType* n1, NodeType* n2) {
809  if (n1 == n2) {
810  return 0;
811  }
812  if (n1->name < n2->name) {
813  return -1;
814  }
815  if (n1->name > n2->name) {
816  return +1;
817  }
818  return (n1 < n2) ? -1 : +1;
819  }
820 
826  static int compare(ArcType* a1, ArcType* a2) {
827  if (a1 == a2) {
828  return 0;
829  }
830  NodeType* n1 = a1->start;
831  NodeType* n2 = a2->start;
832  if (n1 != n2) {
833  return compare(n1, n2);
834  }
835  n1 = a1->finish;
836  n2 = a2->finish;
837  if (n1 != n2) {
838  return compare(n1, n2);
839  }
840  return (a1 < a2) ? -1 : +1;
841  }

930  * Implementation notes: addNode
931  * -----------------------------
932  * The addNode method appears in two forms: one that creates a node
933  * from its name and one that assumes that the client has created
934  * the new node. In each case, the implementation must add the node
935  * the set of nodes for the graph and add the name-to-node association
936  * to the node map.
937  */
938 template <typename NodeType, typename ArcType>
939 NodeType* Graph<NodeType, ArcType>::addNode(const string& name) {
940  NodeType* node = getNode(name);
941  if (node) {
942  return node; // vertex already exists
943  }
944  node = new NodeType();
945  node->arcs = Set<ArcType*>(comparator);
946  node->name = name;
947  return addNode(node);
948 }

1526 template <typename NodeType, typename ArcType>
1527 int Graph<NodeType, ArcType>::graphCompare(const Graph<NodeType, ArcType>& graph2) const {
1528  // optimization: if literally the same graph, return true
1529  if (this == &graph2) {
1530  return 0;
1531  }
1532 
1533  auto itr1 = begin();
1534  auto itr2 = graph2.begin();
1535  auto g1end = end();
1536  auto g2end = graph2.end();
1537 
1538  while (itr1 != g1end && itr2 != g2end) {
1539  // compare each pair of elements from iterators
1540  NodeType* node1 = *itr1;
1541  NodeType* node2 = *itr2;
1542 
1543  // optimization: if literally same node, equal; don't compare
1544  if (node1 != node2) {
1545  // first check names
1546  if (node1->name != node2->name) {
1547  return node1->name.compare(node2->name);
1548  }
1549 
1550  // then check all arcs, pairwise
1551  auto eitr1 = node1->arcs.begin();
1552  auto eitr2 = node2->arcs.begin();
1553  auto e1end = node1->arcs.end();
1554  auto e2end = node2->arcs.end();
1555  while (eitr1 != e1end && eitr2 != e2end) {
1556  ArcType* arc1 = *eitr1;
1557  ArcType* arc2 = *eitr2;
1558 
1559  // optimization: if literally same arc, equal; don't compare
1560  if (arc1 != arc2) {
1561  // first check start vertex names, then end vertex names
1562  if (arc1->start->name != arc2->start->name) {
1563  return arc1->start->name.compare(arc2->start->name);
1564  } else if (arc1->finish->name != arc2->finish->name) {
1565  return arc1->finish->name.compare(arc2->finish->name);
1566  }
1567  }
1568  eitr1++;
1569  eitr2++;
1570  }
1571 
1572  // if we get here, everything from me matched graph2, so either arcs equal,
1573  // or one is shorter than the other (fewer arcs) and is therefore less
1574  if (eitr1 == e1end && eitr2 == e2end) {
1575  // keep going
1576  } else if (eitr1 == e1end) {
1577  return -1;
1578  } else {
1579  return 1;
1580  }
1581  }
1582 
1583  // if we get here, those two vertices and their outbound arcs
1584  // were equal; so advance to next element
1585  itr1++;
1586  itr2++;
1587  }
1588 
1589  // if we get here, everything from me matched graph2, so either equal,
1590  // or one is shorter than the other (fewer vertices) and is therefore less
1591  if (itr1 == g1end && itr2 == g2end) {
1592  return 0;
1593  } else if (itr1 == g1end) {
1594  return -1;
1595  } else {
1596  return 1;
1597  }
1598 }
