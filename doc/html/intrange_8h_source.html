265 /*
266  * An IntRange2D is an iterable contiguous 2-dimensional range of integers.
267  * It can be processed using a for-each loop to emit the range as a series
268  * of Point objects in y-major (default) or x-major order.
269  * The points are not actually all stored in this object, so it does not
270  * require O(WxH) memory usage.
271  */
272 class IntRange2D {
273 private:
274  /*
275  * Internal iterator for looping over a 2D range.
276  */
277  class IntRange2DIterator : public std::iterator<std::input_iterator_tag, Point> {
278  private:
279  const IntRange2D* r;
280  int x;
281  int y;
282 
283  public:
284  IntRange2DIterator(const IntRange2D* r, bool end) {
285  this->r = r;
286  if (end) {
287  this->x = r->maxX() + 1;
288  this->y = r->maxY() + 1;
289  } else {
290  this->x = r->minX();
291  this->y = r->minY();
292  }
293  }
294 
295  IntRange2DIterator(const IntRange2DIterator& itr)
296  : r(itr.r),
297  x(itr.x),
298  y(itr.y) {
299  // empty
300  }
301 
302  IntRange2DIterator& operator ++() {
303  if (r->isYMajor()) {
304  x++;
305  if (x > r->maxX()) {
306  x = r->minX();
307  y++;
308  }
309  } else {
310  y++;
311  if (y > r->maxY()) {
312  y = r->minY();
313  x++;
314  }
315  }
316  if (!r->contains(x, y)) {
317  x = r->maxX() + 1;
318  y = r->maxY() + 1;
319  }
320  return *this;
321  }
322 
323  IntRange2DIterator operator ++(int) {
324  IntRange2DIterator copy(*this);
325  operator++();
326  return copy;
327  }
328 
329  IntRange2DIterator& operator --() {
330  if (r->isYMajor()) {
331  x--;
332  if (x < r->minX()) {
333  x = r->maxX();
334  y--;
335  }
336  } else {
337  y--;
338  if (y < r->minY()) {
339  y = r->maxY();
340  x--;
341  }
342  }
343  return *this;
344  }
345 
346  IntRange2DIterator operator --(int) {
347  IntRange2DIterator copy(*this);
348  operator--();
349  return copy;
350  }

352  bool operator ==(const IntRange2DIterator& rhs) const {
353  return r == rhs.r && x == rhs.x && y == rhs.y;
354  }
355 
356  bool operator !=(const IntRange2DIterator& rhs) const {
357  return !(*this == rhs);
358  }

360  bool operator <(const IntRange2DIterator& rhs) const {
361  if (r != rhs.r) {
362  error("Range2D Iterator::operator <: Iterators are in different ranges");
363  }
364  return x < rhs.x || (x == rhs.x && y < rhs.y);
365  }
366 
367  bool operator <=(const IntRange2DIterator& rhs) const {
368  if (r != rhs.r) {
369  error("Range2D Iterator::operator <=: Iterators are in different ranges");
370  }
371  return (*this < rhs) || (*this == rhs);
372  }
373 
374  bool operator >(const IntRange2DIterator& rhs) const {
375  if (r != rhs.r) {
376  error("Range2D Iterator::operator >: Iterators are in different ranges");
377  }
378  return (rhs < *this);
379  }
380

384  }

386  }
