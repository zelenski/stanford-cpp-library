55  * Class: LinkedList<ValueType>
56  * ----------------------------
57  * This class stores an ordered list of values similar to an array.
58  * It supports traditional array selection using square brackets, but
59  * also supports inserting and deleting elements. It is similar in
60  * function to the STL <code>list</code> type.

743  // loop using iterator to avoid O(N^2) runtime
744  int i = 0;
745  for (const ValueType& element : *this) {
746  if (element == value) {
747  return i;

749  i++;
